## 目录创建命令

```bash
# 创建前端目录结构
mkdir -p src/components/{common,notes,modeling,code,simulation,layout}
mkdir -p src/views
mkdir -p src/stores
mkdir -p src/composables
mkdir -p src/utils
mkdir -p src/types
mkdir -p src/assets/{icons,images}
mkdir -p src/styles

# 创建后端目录结构
mkdir -p src-tauri/src/commands
mkdir -p src-tauri/src/db/migrations
mkdir -p src-tauri/src/models
mkdir -p src-tauri/src/solver

# 创建其他目录
mkdir -p solver/calculix
mkdir -p solver/scripts
mkdir -p docs/{PRD,design,development}
mkdir -p scripts
mkdir -p tests/{unit,integration,e2e}
```
